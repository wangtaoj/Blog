### redis单机

1. 下载redis.conf文件

   ```bash
   wget https://raw.githubusercontent.com/redis/redis/6.2/redis.conf
   ```

2. docker-compose.yml

   ```yaml
   services:
     redis:
       image: redis:6.2
       container_name: redis
       ports:
         - 6379:6379
       volumes:
         - ./redis.conf:/etc/redis.conf
         - ./data:/data
       command: ["redis-server", "/etc/redis.conf"]
   ```

   挂载了配置文件以及数据目录

3. 配置文件修改

   ```properties
   # ip
   bind 0.0.0.0
   # 密码
   requirepass 123456
   ```

   **注: 在docker中运行时，属性daemonize不能设置为yes，否则容器死活启动不了，默认为no**
4. 运行

   运行前注意挂载的配置文件及目录

   ```bash
   docker compose run -d
   ```

### rocketmq单机

1. docker-compose.yml

   ```yaml
   services:
     rocketmq-namesrv:
       image: dyrnq/rocketmq:4.9.4
       container_name: rocketmq-namesrv
       ports:
         - 9876:9876
       volumes:
         - ./mqnamesrv/logs:/home/rocketmq/logs
         - ./mqnamesrv/store:/home/rocketmq/store
       environment:
         TZ: "Asia/Shanghai"
         JAVA_OPT_EXT: "-Duser.home=/home/rocketmq -Xms512m -Xmx512m -Xmn128m -XX:MetaspaceSize=128m -XX:MaxMetaspaceSize=128m"
       command: ["mqnamesrv"]
       networks:
           rocketmq:
             aliases:
               - rocketmq-namesrv
   
     rocketmq-broker:
       image: dyrnq/rocketmq:4.9.4
       container_name: rocketmq-broker
       ports:
         - 10909:10909
         - 10911:10911
       volumes:
         - ./mqbroker/logs:/home/rocketmq/logs
         - ./mqbroker/store:/home/rocketmq/store
         - ./mqbroker/conf/broker.conf:/etc/rocketmq/broker.conf
       environment:
           TZ: "Asia/Shanghai"
           JAVA_OPT_EXT: "-Duser.home=/home/rocketmq -Xms512m -Xmx512m -Xmn128m -XX:MetaspaceSize=128m -XX:MaxMetaspaceSize=128m"
       command: ["mqbroker", "-n", "rocketmq-namesrv:9876", "-c", "/etc/rocketmq/broker.conf"]
       depends_on:
         - rocketmq-namesrv
       networks:
         rocketmq:
           aliases:
             - rocketmq-broker
   
     rocketmq-dashboard:
       image: schnell18/rocketmq-dashboard:1.0.0-alpine
       container_name: rocketmq-dashboard
       ports:
         - 7080:8080
       volumes:
         - ./mqdashboard/logs:/rocketmq-console/logs
         - ./mqdashboard/data:/rocketmq-console/data
       environment:
         ROCKETMQ_CONFIG_NAMESRVADDR: rocketmq-namesrv:9876
       depends_on:
         - rocketmq-namesrv
       networks:
         rocketmq:
           aliases:
             - rocketmq-dashboard
       
   
   networks:
     rocketmq:
       name: rocketmq
       driver: bridge
   ```

   # 集群名字
   brokerClusterName=DefaultCluster
   
   # broker名字
   brokerName=broker-a
   
   # 0表示Master >0表示Slave
   brokerId=0
   
   # nameServer地址，分号分割(启动时指定，此处注释)
   # namesrvAddr=rocketmq-namesrv:9876
   
   # 在发送消息时，自动创建服务器不存在的topic，默认创建的队列数
   defaultTopicQueueNums=4
   
   # 文件删除时间点，默认凌晨4点
   deleteWhen=04
   
   # 文件保留时长，默认48小时
   fileReservedTime=48
   
   # Broker的角色
   # ASYNC_MASTER 异步复制Master
   # SYNC_MASTER 同步双写Master
   # SLAVE
   brokerRole=ASYNC_MASTER
   
   # 刷盘方式
   # ASYNC_FLUSH 异步刷盘
   # SYNC_FLUSH 同步刷盘
   flushDiskType=ASYNC_FLUSH
   
   # 启动IP,如果 docker 报 com.alibaba.rocketmq.remoting.exception.RemotingConnectException: connect to <192.168.0.120:10909> failed
   # 解决方式1 加上一句producer.setVipChannelEnabled(false);，解决方式2 brokerIP1 设置宿主机IP，不要使用docker 内部IP
